44CmdCardSvDebitOrUndebit::CmdCardSvDebitOrUndebit(
45 const bool isDebitCommand,
46 const std::shared_ptr<CalypsoCardAdapter> calypsoCard,
47 const int amount,
48 const std::vector<uint8_t>& date,
49 const std::vector<uint8_t>& time,
50 const bool isExtendedModeAllowed)
51: AbstractCardCommand(isDebitCommand ? CalypsoCardCommand::SV_DEBIT :
52 CalypsoCardCommand::SV_UNDEBIT,
53 0,
54 calypsoCard),
55 /* Keeps a copy of these fields until the command is finalized */
56 mIsExtendedModeAllowed(isExtendedModeAllowed)
57{
58 /*
59 * @see Calypso Layer ID 8.02 (200108)
60 * CL-SV-DEBITVAL.1
61 */
62 if (amount < 0 || amount > 32767) {
63
64 throw IllegalArgumentException("Amount is outside allowed boundaries (0 <= amount <= " \
65 "32767)");
66 }
67
68 if (date.empty() || time.empty()) {
69
70 throw IllegalArgumentException("date and time cannot be null");
71 }
72
73 if (date.size() != 2 || time.size() != 2) {
74
75 throw IllegalArgumentException("date and time must be 2-byte arrays");
76 }
77
78 /*
79 * Handle the dataIn size with signatureHi length according to card product type (3.2 rev have a
80 * 10-byte signature)
81 */
82 mDataIn = std::vector<uint8_t>(15 + (isExtendedModeAllowed ? 10 : 5));
83
84 /* mDataIn[0] will be filled in at the finalization phase */
85 const short amountShort = isDebitCommand ?
86 static_cast<short>(-amount) : static_cast<short>(amount);
87 ByteArrayUtil::copyBytes(amountShort, mDataIn, 1, 2);
88 mDataIn[3] = date[0];
89 mDataIn[4] = date[1];
90 mDataIn[5] = time[0];
91 mDataIn[6] = time[1];
92 mDataIn[7] = calypsoCard->getSvKvc();
93 /* mDataIn[8]..dataIn[8+7+sigLen] will be filled in at the finalization phase */
94}

96void CmdCardSvDebitOrUndebit::finalizeCommand(
97 const std::vector<uint8_t>& debitOrUndebitComplementaryData)
98{
99 if ((mIsExtendedModeAllowed && debitOrUndebitComplementaryData.size() != 20) ||
100 (!mIsExtendedModeAllowed && debitOrUndebitComplementaryData.size() != 15)) {
101
102 throw IllegalArgumentException("Bad SV prepare load data length.");
103 }
104
105 const uint8_t p1 = debitOrUndebitComplementaryData[4];
106 const uint8_t p2 = debitOrUndebitComplementaryData[5];
107
108 mDataIn[0] = debitOrUndebitComplementaryData[6];
109 System::arraycopy(debitOrUndebitComplementaryData, 0, mDataIn, 8, 4);
110 System::arraycopy(debitOrUndebitComplementaryData, 7, mDataIn, 12, 3);
111 System::arraycopy(debitOrUndebitComplementaryData,
112 10,
113 mDataIn,
114 15,
115 debitOrUndebitComplementaryData.size() - 10);
116
117 const uint8_t cardClass = getCalypsoCard()->getCardClass() == CalypsoCardClass::LEGACY ?
118 CalypsoCardClass::LEGACY_STORED_VALUE.getValue() :
119 CalypsoCardClass::ISO.getValue();
120
121 auto apduRequest = std::make_shared<ApduRequestAdapter>(
122 ApduUtil::build(cardClass,
123 getCommandRef().getInstructionByte(),
124 p1,
125 p2,
126 mDataIn));
127 apduRequest->addSuccessfulStatusWord(SW_POSTPONED_DATA);
128 setApduRequest(apduRequest);
129}

131const std::vector<uint8_t> CmdCardSvDebitOrUndebit::getSvDebitOrUndebitData() const
132{
133 std::vector<uint8_t> svDebitOrUndebitData(12);
134
135 svDebitOrUndebitData[0] = getCommandRef().getInstructionByte();
136
137 /*
138 * svDebitOrUndebitData[1,2] / P1P2 not set because ignored
139 * Lc is 5 bytes longer in product type 3.2
140 */
141 svDebitOrUndebitData[3] = mIsExtendedModeAllowed ? 0x19 : 0x14;
142
143 /* Appends the fixed part of dataIn */
144 System::arraycopy(mDataIn, 0, svDebitOrUndebitData, 4, 8);
145
146 return svDebitOrUndebitData;
147}

154void CmdCardSvDebitOrUndebit::parseApduResponse(const std::shared_ptr<ApduResponseApi> apduResponse)
155{
156 AbstractCardCommand::parseApduResponse(apduResponse);
157
158 const std::vector<uint8_t> dataOut = apduResponse->getDataOut();
159
160 if (dataOut.size() != 0 && dataOut.size() != 3 && dataOut.size() != 6) {
161
162 throw IllegalStateException("Bad length in response to SV Debit/Undebit command.");
163 }
164
165 getCalypsoCard()->setSvOperationSignature(apduResponse->getDataOut());
166}

189 std::make_shared<StatusProperties>("Preconditions not satisfied.",
190 typeid(CalypsoSamAccessForbiddenException))});
191 m.insert({0x6988,
192 std::make_shared<StatusProperties>("Incorrect signatureHi.",
193 typeid(CardSecurityDataException))});
