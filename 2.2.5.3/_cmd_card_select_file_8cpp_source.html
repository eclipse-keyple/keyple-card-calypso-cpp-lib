75void CmdCardSelectFile::buildCommand(const CalypsoCardClass calypsoCardClass,
76 const SelectFileControl selectFileControl)
77{
78 const uint8_t cla = calypsoCardClass.getValue();
79 uint8_t p1;
80 uint8_t p2;
81 const std::vector<uint8_t> selectData = {0x00, 0x00};
82
83 switch (selectFileControl) {
84
85 case SelectFileControl::FIRST_EF:
86 p1 = 0x02;
87 p2 = 0x00;
88 break;
89
90 case SelectFileControl::NEXT_EF:
91 p1 = 0x02;
92 p2 = 0x02;
93 break;
94
95 case SelectFileControl::CURRENT_DF:
96 /* CL-KEY-KIFSF.1 */
97 p1 = 0x09;
98 p2 = 0x00;
99 break;
100
101 default:
102 throw IllegalStateException("Unsupported selectFileControl parameter " \
103 "FIXME: selectFileControl.name()");
104 }
105
106 setApduRequest(
107 std::make_shared<ApduRequestAdapter>(
108 ApduUtil::build(cla, mCommand.getInstructionByte(), p1, p2, selectData, 0x00)));
109
110
111 addSubName("SELECTIONCONTROL FIXME: selectFileControl");
112}

114void CmdCardSelectFile::buildCommand(const CalypsoCardClass calypsoCardClass,
115 const CalypsoCard::ProductType productType,
116 const uint16_t lid)
117{
118 /*
119 * Handle the REV1 case
120 * CL-KEY-KIFSF.1
121 * If legacy and rev2 then 02h else if legacy then 08h else 09h
122 */
123 uint8_t p1;
124
125 if (calypsoCardClass == CalypsoCardClass::LEGACY &&
126 productType == CalypsoCard::ProductType::PRIME_REVISION_2) {
127
128 p1 = 0x02;
129
130 } else if (calypsoCardClass == CalypsoCardClass::LEGACY) {
131
132 p1 = 0x08;
133
134 } else {
135
136 p1 = 0x09;
137 }
138
139 const std::vector<uint8_t> dataIn = ByteArrayUtil::extractBytes(lid, 2);
140
141 setApduRequest(
142 std::make_shared<ApduRequestAdapter>(
143 ApduUtil::build(calypsoCardClass.getValue(),
144 mCommand.getInstructionByte(),
145 p1,
146 0x00,
147 dataIn,
148 0x00)));
149
150 addSubName("LID=" + HexUtil::toHex(dataIn));
151}

160void CmdCardSelectFile::parseProprietaryInformation(
161 const std::vector<uint8_t>& dataOut,
162 const std::shared_ptr<CalypsoCardAdapter> calypsoCard)
163{
164 const std::vector<uint8_t>& proprietaryInformation = getProprietaryInformation(dataOut);
165 const uint8_t sfi = proprietaryInformation[CalypsoCardConstant::SEL_SFI_OFFSET];
166 const uint8_t fileType = proprietaryInformation[CalypsoCardConstant::SEL_TYPE_OFFSET];
167
168 if (fileType == CalypsoCardConstant::FILE_TYPE_MF ||
169 fileType == CalypsoCardConstant::FILE_TYPE_DF) {
170
171 std::shared_ptr<DirectoryHeader> directoryHeader =
172 createDirectoryHeader(proprietaryInformation, calypsoCard);
173 calypsoCard->setDirectoryHeader(directoryHeader);
174
175 } else if (fileType == CalypsoCardConstant::FILE_TYPE_EF) {
176
177 std::shared_ptr<FileHeaderAdapter> fileHeader =
178 createFileHeader(proprietaryInformation, calypsoCard);
179 calypsoCard->setFileHeader(sfi, fileHeader);
180
181 } else {
182
183 throw IllegalStateException(StringUtils::format("Unknown file type: %02Xh", fileType));
184 }
185}

192const std::vector<uint8_t> CmdCardSelectFile::getProprietaryInformation(
193 const std::vector<uint8_t>& dataOut)
194{
195 std::vector<uint8_t> proprietaryInformation;
196 const std::map<const int, const std::vector<uint8_t>> tags =
197 BerTlvUtil::parseSimple(dataOut, true);
198
199 const auto& it = tags.find(TAG_PROPRIETARY_INFORMATION);
200
201 if (it == tags.end()) {
202
203 throw IllegalStateException("Proprietary information: tag not found.");
204 }
205
206 proprietaryInformation = it->second;
207
208 Assert::getInstance().isEqual(proprietaryInformation.size(), 23, "proprietaryInformation");
209
210 return proprietaryInformation;
211}

213const std::shared_ptr<DirectoryHeader> CmdCardSelectFile::createDirectoryHeader(
214 const std::vector<uint8_t>& proprietaryInformation,
215 const std::shared_ptr<CalypsoCardAdapter> calypsoCard)
216
217{
218 std::vector<uint8_t> accessConditions(CalypsoCardConstant::SEL_AC_LENGTH);
219 System::arraycopy(proprietaryInformation,
220 CalypsoCardConstant::SEL_AC_OFFSET,
221 accessConditions,
222 0,
223 CalypsoCardConstant::SEL_AC_LENGTH);
224
225 std::vector<uint8_t> keyIndexes(CalypsoCardConstant::SEL_NKEY_LENGTH);
226 System::arraycopy(proprietaryInformation,
227 CalypsoCardConstant::SEL_NKEY_OFFSET,
228 keyIndexes,
229 0,
230 CalypsoCardConstant::SEL_NKEY_LENGTH);
231
232 const uint8_t dfStatus = proprietaryInformation[CalypsoCardConstant::SEL_DF_STATUS_OFFSET];
233
234 const int lidOffset =
235 calypsoCard->getProductType() == CalypsoCard::ProductType::PRIME_REVISION_2 ?
236 CalypsoCardConstant::SEL_LID_OFFSET_REV2 : CalypsoCardConstant::SEL_LID_OFFSET;
237
238 const uint8_t lid = ByteArrayUtil::extractShort(proprietaryInformation, lidOffset);
239
240 return DirectoryHeaderAdapter::builder()
241 ->lid(lid)
242 .accessConditions(accessConditions)
243 .keyIndexes(keyIndexes)
244 .dfStatus(dfStatus)
245 .kvc(WriteAccessLevel::PERSONALIZATION,
246 proprietaryInformation[CalypsoCardConstant::SEL_KVCS_OFFSET])
247 .kvc(WriteAccessLevel::LOAD,
248 proprietaryInformation[CalypsoCardConstant::SEL_KVCS_OFFSET + 1])
249 .kvc(WriteAccessLevel::DEBIT,
250 proprietaryInformation[CalypsoCardConstant::SEL_KVCS_OFFSET + 2])
251 .kif(WriteAccessLevel::PERSONALIZATION,
252 proprietaryInformation[CalypsoCardConstant::SEL_KIFS_OFFSET])
253 .kif(WriteAccessLevel::LOAD,
254 proprietaryInformation[CalypsoCardConstant::SEL_KIFS_OFFSET + 1])
255 .kif(WriteAccessLevel::DEBIT,
256 proprietaryInformation[CalypsoCardConstant::SEL_KIFS_OFFSET + 2])
257 .build();
258}

260const std::shared_ptr<FileHeaderAdapter> CmdCardSelectFile::createFileHeader(
261 const std::vector<uint8_t>& proprietaryInformation,
262 const std::shared_ptr<CalypsoCardAdapter> calypsoCard)
263{
264 const ElementaryFile::Type fileType =
265 getEfTypeFromCardValue(proprietaryInformation[CalypsoCardConstant::SEL_EF_TYPE_OFFSET]);
266
267 int recordSize;
268 int recordsNumber;
269
270 if (fileType == ElementaryFile::Type::BINARY) {
271
272 recordSize = ByteArrayUtil::extractInt(proprietaryInformation,
273 CalypsoCardConstant::SEL_REC_SIZE_OFFSET,
274 2,
275 false);
276 recordsNumber = 1;
277
278 } else {
279
280 recordSize = proprietaryInformation[CalypsoCardConstant::SEL_REC_SIZE_OFFSET];
281 recordsNumber = proprietaryInformation[CalypsoCardConstant::SEL_NUM_REC_OFFSET];
282 }
283
284 std::vector<uint8_t> accessConditions(CalypsoCardConstant::SEL_AC_LENGTH);
285 System::arraycopy(proprietaryInformation,
286 CalypsoCardConstant::SEL_AC_OFFSET,
287 accessConditions,
288 0,
289 CalypsoCardConstant::SEL_AC_LENGTH);
290
291 std::vector<uint8_t> keyIndexes(CalypsoCardConstant::SEL_NKEY_LENGTH);
292 System::arraycopy(proprietaryInformation,
293 CalypsoCardConstant::SEL_NKEY_OFFSET,
294 keyIndexes,
295 0,
296 CalypsoCardConstant::SEL_NKEY_LENGTH);
297
298 const uint8_t dfStatus = proprietaryInformation[CalypsoCardConstant::SEL_DF_STATUS_OFFSET];
299
300 const uint16_t sharedReference =
301 ByteArrayUtil::extractShort(proprietaryInformation,
302 CalypsoCardConstant::SEL_DATA_REF_OFFSET);
303
304 const int lidOffset =
305 calypsoCard->getProductType() == CalypsoCard::ProductType::PRIME_REVISION_2 ?
306 CalypsoCardConstant::SEL_LID_OFFSET_REV2 : CalypsoCardConstant::SEL_LID_OFFSET;
307
308 const uint16_t lid = ByteArrayUtil::extractShort(proprietaryInformation, lidOffset);
309
310 return FileHeaderAdapter::builder()
311 ->lid(lid)
312 .recordsNumber(recordsNumber)
313 .recordSize(recordSize)
314 .type(fileType)
315 .accessConditions(Arrays::copyOf(accessConditions, accessConditions.size()))
316 .keyIndexes(Arrays::copyOf(keyIndexes, keyIndexes.size()))
317 .dfStatus(dfStatus)
318 .sharedReference(sharedReference)
319 .build();
320}

322ElementaryFile::Type CmdCardSelectFile::getEfTypeFromCardValue(const uint8_t efType)
323{
324 if (efType == CalypsoCardConstant::EF_TYPE_BINARY) {
325
326 return ElementaryFile::Type::BINARY;
327
328 } else if (efType == CalypsoCardConstant::EF_TYPE_LINEAR) {
329
330 return ElementaryFile::Type::LINEAR;
331
332 } else if (efType == CalypsoCardConstant::EF_TYPE_CYCLIC) {
333
334 return ElementaryFile::Type::CYCLIC;
335
336 } else if (efType == CalypsoCardConstant::EF_TYPE_SIMULATED_COUNTERS) {
337
338 return ElementaryFile::Type::SIMULATED_COUNTERS;
339
340 } else if (efType == CalypsoCardConstant::EF_TYPE_COUNTERS) {
341
342 return ElementaryFile::Type::COUNTERS;
343
344 } else {
345
346 throw IllegalStateException("Unknown EF Type: " + std::to_string(efType));
347 }
