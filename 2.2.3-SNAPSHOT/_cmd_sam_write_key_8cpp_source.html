40CmdSamWriteKey::CmdSamWriteKey(
41 const std::shared_ptr<CalypsoSamAdapter> calypsoSam,
42 const uint8_t writingMode,
43 const uint8_t keyReference,
44 const std::vector<uint8_t>& keyData)
45: AbstractSamCommand(mCommand, 0, calypsoSam)
46{
47 const uint8_t cla = SamUtilAdapter::getClassByte(calypsoSam->getProductType());
48
49 if (keyData.empty()) {
50
51 throw IllegalArgumentException("Key data null!");
52 }
53
54 if (keyData.size() < 48 && keyData.size() > 80) {
55
56 throw IllegalArgumentException("Key data should be between 40 and 80 bytes long!");
57 }
58
59 setApduRequest(
60 std::make_shared<ApduRequestAdapter>(
61 ApduUtil::build(cla,
62 mCommand.getInstructionByte(),
63 writingMode,
64 keyReference,
65 keyData)));
66}
