107const std::map<const std::shared_ptr<FileHeaderAdapter>, const uint8_t>
108 CmdCardGetDataEfList::getEfHeaders() const
109{
110 const std::vector<uint8_t> rawList = getApduResponse()->getDataOut();
111 std::map<const std::shared_ptr<FileHeaderAdapter>, const uint8_t> fileHeaderToSfiMap;
112 const int nbFiles = rawList[1] / DESCRIPTOR_TAG_LENGTH;
113
114 for (int i = 0; i < nbFiles; i++) {
115 fileHeaderToSfiMap.insert({
116 createFileHeader(
117 Arrays::copyOfRange(
118 rawList,
119 DESCRIPTORS_OFFSET + (i * DESCRIPTOR_TAG_LENGTH) + DESCRIPTOR_DATA_OFFSET,
120 DESCRIPTORS_OFFSET
121 + (i * DESCRIPTOR_TAG_LENGTH)
122 + DESCRIPTOR_DATA_OFFSET
123 + DESCRIPTOR_DATA_LENGTH)),
124 rawList[
125 DESCRIPTORS_OFFSET
126 + (i * DESCRIPTOR_TAG_LENGTH)
127 + DESCRIPTOR_DATA_OFFSET
128 + DESCRIPTOR_DATA_SFI_OFFSET]});
129 }
130
131 return fileHeaderToSfiMap;
132}

134const std::shared_ptr<FileHeaderAdapter> CmdCardGetDataEfList::createFileHeader(
135 const std::vector<uint8_t>& efDescriptorByteArray) const
136{
137 ElementaryFile::Type efType;
138
139 if (efDescriptorByteArray[3] == CalypsoCardConstant::EF_TYPE_LINEAR) {
140
141 efType = ElementaryFile::Type::LINEAR;
142
143 } else if (efDescriptorByteArray[3] == CalypsoCardConstant::EF_TYPE_CYCLIC) {
144
145 efType = ElementaryFile::Type::CYCLIC;
146
147 } else if (efDescriptorByteArray[3] == CalypsoCardConstant::EF_TYPE_COUNTERS) {
148
149 efType = ElementaryFile::Type::COUNTERS;
150
151 } else if (efDescriptorByteArray[3] == CalypsoCardConstant::EF_TYPE_BINARY) {
152
153 efType = ElementaryFile::Type::BINARY;
154
155 } else if (efDescriptorByteArray[3] == CalypsoCardConstant::EF_TYPE_SIMULATED_COUNTERS) {
156
157 efType = ElementaryFile::Type::SIMULATED_COUNTERS;
158
159 } else {
160
161 throw IllegalStateException("Unexpected EF type");
162 }
163
164 return FileHeaderAdapter::builder()
165 ->lid(ByteArrayUtil::extractShort(efDescriptorByteArray, 0))
166 .type(efType)
167 .recordSize(efDescriptorByteArray[4])
168 .recordsNumber(efDescriptorByteArray[5])
169 .build();
170}
