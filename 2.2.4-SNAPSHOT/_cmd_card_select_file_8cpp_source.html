74void CmdCardSelectFile::buildCommand(const CalypsoCardClass calypsoCardClass,
75 const SelectFileControl selectFileControl)
76{
77 const uint8_t cla = calypsoCardClass.getValue();
78 uint8_t p1;
79 uint8_t p2;
80 const std::vector<uint8_t> selectData = {0x00, 0x00};
81
82 switch (selectFileControl) {
83
84 case SelectFileControl::FIRST_EF:
85 p1 = 0x02;
86 p2 = 0x00;
87 break;
88
89 case SelectFileControl::NEXT_EF:
90 p1 = 0x02;
91 p2 = 0x02;
92 break;
93
94 case SelectFileControl::CURRENT_DF:
95 /* CL-KEY-KIFSF.1 */
96 p1 = 0x09;
97 p2 = 0x00;
98 break;
99
100 default:
101 throw IllegalStateException("Unsupported selectFileControl parameter " \
102 "FIXME: selectFileControl.name()");
103 }
104
105 setApduRequest(
106 std::make_shared<ApduRequestAdapter>(
107 ApduUtil::build(cla, mCommand.getInstructionByte(), p1, p2, selectData, 0x00)));
108
109
110 addSubName("SELECTIONCONTROL FIXME: selectFileControl");
111}

113void CmdCardSelectFile::buildCommand(const CalypsoCardClass calypsoCardClass,
114 const CalypsoCard::ProductType productType,
115 const uint16_t lid)
116{
117 /*
118 * Handle the REV1 case
119 * CL-KEY-KIFSF.1
120 * If legacy and rev2 then 02h else if legacy then 08h else 09h
121 */
122 uint8_t p1;
123
124 if (calypsoCardClass == CalypsoCardClass::LEGACY &&
125 productType == CalypsoCard::ProductType::PRIME_REVISION_2) {
126
127 p1 = 0x02;
128
129 } else if (calypsoCardClass == CalypsoCardClass::LEGACY) {
130
131 p1 = 0x08;
132
133 } else {
134
135 p1 = 0x09;
136 }
137
138 const std::vector<uint8_t> dataIn = {static_cast<uint8_t>((lid >> 8) & 0xFF),
139 static_cast<uint8_t>(lid & 0xFF)};
140
141 setApduRequest(
142 std::make_shared<ApduRequestAdapter>(
143 ApduUtil::build(calypsoCardClass.getValue(),
144 mCommand.getInstructionByte(),
145 p1,
146 0x00,
147 dataIn,
148 0x00)));
149
150 addSubName("LID=" + HexUtil::toHex(dataIn));
151}

160void CmdCardSelectFile::parseProprietaryInformation(
161 const std::vector<uint8_t>& dataOut,
162 const std::shared_ptr<CalypsoCardAdapter> calypsoCard)
163{
164 const std::vector<uint8_t>& proprietaryInformation = getProprietaryInformation(dataOut);
165 const uint8_t sfi = proprietaryInformation[CalypsoCardConstant::SEL_SFI_OFFSET];
166 const uint8_t fileType = proprietaryInformation[CalypsoCardConstant::SEL_TYPE_OFFSET];
167
168 if (fileType == CalypsoCardConstant::FILE_TYPE_MF ||
169 fileType == CalypsoCardConstant::FILE_TYPE_DF) {
170
171 std::shared_ptr<DirectoryHeader> directoryHeader =
172 createDirectoryHeader(proprietaryInformation, calypsoCard);
173 calypsoCard->setDirectoryHeader(directoryHeader);
174
175 } else if (fileType == CalypsoCardConstant::FILE_TYPE_EF) {
176
177 std::shared_ptr<FileHeaderAdapter> fileHeader =
178 createFileHeader(proprietaryInformation, calypsoCard);
179 calypsoCard->setFileHeader(sfi, fileHeader);
180
181 } else {
182
183 throw IllegalStateException(StringUtils::format("Unknown file type: %02Xh", fileType));
184 }
185}

192const std::vector<uint8_t> CmdCardSelectFile::getProprietaryInformation(
193 const std::vector<uint8_t>& dataOut)
194{
195 std::vector<uint8_t> proprietaryInformation;
196 const std::map<const int, const std::vector<uint8_t>> tags =
197 BerTlvUtil::parseSimple(dataOut, true);
198
199 const auto& it = tags.find(TAG_PROPRIETARY_INFORMATION);
200
201 if (it == tags.end()) {
202
203 throw IllegalStateException("Proprietary information: tag not found.");
204 }
205
206 proprietaryInformation = it->second;
207
208 Assert::getInstance().isEqual(proprietaryInformation.size(), 23, "proprietaryInformation");
209
210 return proprietaryInformation;
211}

213const std::shared_ptr<DirectoryHeader> CmdCardSelectFile::createDirectoryHeader(
214 const std::vector<uint8_t>& proprietaryInformation,
215 const std::shared_ptr<CalypsoCardAdapter> calypsoCard)
216
217{
218 std::vector<uint8_t> accessConditions(CalypsoCardConstant::SEL_AC_LENGTH);
219 System::arraycopy(proprietaryInformation,
220 CalypsoCardConstant::SEL_AC_OFFSET,
221 accessConditions,
222 0,
223 CalypsoCardConstant::SEL_AC_LENGTH);
224
225 std::vector<uint8_t> keyIndexes(CalypsoCardConstant::SEL_NKEY_LENGTH);
226 System::arraycopy(proprietaryInformation,
227 CalypsoCardConstant::SEL_NKEY_OFFSET,
228 keyIndexes,
229 0,
230 CalypsoCardConstant::SEL_NKEY_LENGTH);
231
232 const uint8_t dfStatus = proprietaryInformation[CalypsoCardConstant::SEL_DF_STATUS_OFFSET];
233
234 const int lidOffset =
235 calypsoCard->getProductType() == CalypsoCard::ProductType::PRIME_REVISION_2 ?
236 CalypsoCardConstant::SEL_LID_OFFSET_REV2 : CalypsoCardConstant::SEL_LID_OFFSET;
237
238 const uint8_t lid = ((proprietaryInformation[lidOffset] << 8) & 0xff00) |
239 ((proprietaryInformation[lidOffset + 1] & 0x00ff));
240
241 return DirectoryHeaderAdapter::builder()
242 ->lid(lid)
243 .accessConditions(accessConditions)
244 .keyIndexes(keyIndexes)
245 .dfStatus(dfStatus)
246 .kvc(WriteAccessLevel::PERSONALIZATION,
247 proprietaryInformation[CalypsoCardConstant::SEL_KVCS_OFFSET])
248 .kvc(WriteAccessLevel::LOAD,
249 proprietaryInformation[CalypsoCardConstant::SEL_KVCS_OFFSET + 1])
250 .kvc(WriteAccessLevel::DEBIT,
251 proprietaryInformation[CalypsoCardConstant::SEL_KVCS_OFFSET + 2])
252 .kif(WriteAccessLevel::PERSONALIZATION,
253 proprietaryInformation[CalypsoCardConstant::SEL_KIFS_OFFSET])
254 .kif(WriteAccessLevel::LOAD,
255 proprietaryInformation[CalypsoCardConstant::SEL_KIFS_OFFSET + 1])
256 .kif(WriteAccessLevel::DEBIT,
257 proprietaryInformation[CalypsoCardConstant::SEL_KIFS_OFFSET + 2])
258 .build();
259}

261const std::shared_ptr<FileHeaderAdapter> CmdCardSelectFile::createFileHeader(
262 const std::vector<uint8_t>& proprietaryInformation,
263 const std::shared_ptr<CalypsoCardAdapter> calypsoCard)
264{
265 const ElementaryFile::Type fileType =
266 getEfTypeFromCardValue(proprietaryInformation[CalypsoCardConstant::SEL_EF_TYPE_OFFSET]);
267
268 int recordSize;
269 int recordsNumber;
270
271 if (fileType == ElementaryFile::Type::BINARY) {
272
273 recordSize =
274 ((proprietaryInformation[CalypsoCardConstant::SEL_REC_SIZE_OFFSET] << 8) & 0x0000ff00) |
275 (proprietaryInformation[CalypsoCardConstant::SEL_NUM_REC_OFFSET] & 0x000000ff);
276 recordsNumber = 1;
277
278 } else {
279
280 recordSize = proprietaryInformation[CalypsoCardConstant::SEL_REC_SIZE_OFFSET];
281 recordsNumber = proprietaryInformation[CalypsoCardConstant::SEL_NUM_REC_OFFSET];
282 }
283
284 std::vector<uint8_t> accessConditions(CalypsoCardConstant::SEL_AC_LENGTH);
285 System::arraycopy(proprietaryInformation,
286 CalypsoCardConstant::SEL_AC_OFFSET,
287 accessConditions,
288 0,
289 CalypsoCardConstant::SEL_AC_LENGTH);
290
291 std::vector<uint8_t> keyIndexes(CalypsoCardConstant::SEL_NKEY_LENGTH);
292 System::arraycopy(proprietaryInformation,
293 CalypsoCardConstant::SEL_NKEY_OFFSET,
294 keyIndexes,
295 0,
296 CalypsoCardConstant::SEL_NKEY_LENGTH);
297
298 const uint8_t dfStatus = proprietaryInformation[CalypsoCardConstant::SEL_DF_STATUS_OFFSET];
299
300 const uint16_t sharedReference =
301 ((proprietaryInformation[CalypsoCardConstant::SEL_DATA_REF_OFFSET] << 8) & 0xff00) |
302 (proprietaryInformation[CalypsoCardConstant::SEL_DATA_REF_OFFSET + 1] & 0x00ff);
303
304 const int lidOffset =
305 calypsoCard->getProductType() == CalypsoCard::ProductType::PRIME_REVISION_2 ?
306 CalypsoCardConstant::SEL_LID_OFFSET_REV2 : CalypsoCardConstant::SEL_LID_OFFSET;
307
308 const uint16_t lid = ((proprietaryInformation[lidOffset] << 8) & 0xff00) |
309 (proprietaryInformation[lidOffset + 1] & 0x00ff);
310
311 return FileHeaderAdapter::builder()
312 ->lid(lid)
313 .recordsNumber(recordsNumber)
314 .recordSize(recordSize)
315 .type(fileType)
316 .accessConditions(Arrays::copyOf(accessConditions, accessConditions.size()))
317 .keyIndexes(Arrays::copyOf(keyIndexes, keyIndexes.size()))
318 .dfStatus(dfStatus)
319 .sharedReference(sharedReference)
320 .build();
321}

323ElementaryFile::Type CmdCardSelectFile::getEfTypeFromCardValue(const uint8_t efType)
324{
325 if (efType == CalypsoCardConstant::EF_TYPE_BINARY) {
326
327 return ElementaryFile::Type::BINARY;
328
329 } else if (efType == CalypsoCardConstant::EF_TYPE_LINEAR) {
330
331 return ElementaryFile::Type::LINEAR;
332
333 } else if (efType == CalypsoCardConstant::EF_TYPE_CYCLIC) {
334
335 return ElementaryFile::Type::CYCLIC;
336
337 } else if (efType == CalypsoCardConstant::EF_TYPE_SIMULATED_COUNTERS) {
338
339 return ElementaryFile::Type::SIMULATED_COUNTERS;
340
341 } else if (efType == CalypsoCardConstant::EF_TYPE_COUNTERS) {
342
343 return ElementaryFile::Type::COUNTERS;
344
345 } else {
346
347 throw IllegalStateException("Unknown EF Type: " + efType);
348 }
