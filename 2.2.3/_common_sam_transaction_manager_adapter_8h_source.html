171 SamTransactionManager& prepareComputeSignature(const any data) override
172 {
173 /* C++: careful, code is a little bit different from Java because of any_cast flow */
174 try {
175
176 /* Basic signature */
177 auto dataAdapter =
178 any_cast<std::shared_ptr<BasicSignatureComputationDataAdapter>>(data);
179
180 Assert::getInstance().notNull(dataAdapter, MSG_INPUT_OUTPUT_DATA)
181 .isInRange(dataAdapter->getData().size(),
182 1,
183 208,
184 "length of data to sign")
185 .isTrue(dataAdapter->getData().size() % 8 == 0,
186 "length of data to sign is a multiple of 8")
187 .isInRange(dataAdapter->getSignatureSize(),
188 1,
189 8,
190 MSG_SIGNATURE_SIZE)
191 .isTrue(dataAdapter->isKeyDiversifierSet() == false ||
192 (dataAdapter->getKeyDiversifier().size() >= 1 &&
193 dataAdapter->getKeyDiversifier().size() <= 8),
194 MSG_KEY_DIVERSIFIER_SIZE_IS_IN_RANGE_1_8);
195
196 prepareSelectDiversifierIfNeeded(dataAdapter->getKeyDiversifier());
197 mSamCommands.push_back(std::make_shared<CmdSamDataCipher>(mSam, dataAdapter, nullptr));
198
199 return *this;
200
201 } catch (const std::exception& e) {
202
203 (void)e;
204 }
205
206 try {
207
208 /* Traceable signature */
209 auto dataAdapter =
210 any_cast<std::shared_ptr<TraceableSignatureComputationDataAdapter>>(data);
211
212 Assert::getInstance().notNull(dataAdapter, MSG_INPUT_OUTPUT_DATA)
213 .isInRange(dataAdapter->getData().size(),
214 1,
215 dataAdapter->isSamTraceabilityMode() ? 206 : 208,
216 "length of data to sign")
217 .isInRange(dataAdapter->getSignatureSize(),
218 1,
219 8,
220 MSG_SIGNATURE_SIZE)
221 .isTrue(!dataAdapter->isSamTraceabilityMode() ||
222 (dataAdapter->getTraceabilityOffset() >= 0 &&
223 dataAdapter->getTraceabilityOffset() <=
224 static_cast<int>((dataAdapter->getData().size() * 8) -
225 (dataAdapter->isPartialSamSerialNumber() ?
226 7 * 8 : 8 * 8))),
227 "traceability offset is in range [0.." +
228 std::to_string(((dataAdapter->getData().size() * 8) -
229 (dataAdapter->isPartialSamSerialNumber() ? 7 * 8 : 8 * 8))) +
230 "]")
231 .isTrue(dataAdapter->isKeyDiversifierSet() == false ||
232 (dataAdapter->getKeyDiversifier().size() >= 1 &&
233 dataAdapter->getKeyDiversifier().size() <= 8),
234 MSG_KEY_DIVERSIFIER_SIZE_IS_IN_RANGE_1_8);
235
236 prepareSelectDiversifierIfNeeded(dataAdapter->getKeyDiversifier());
237 mSamCommands.push_back(
238 std::make_shared<CmdSamPsoComputeSignature>(mSam, dataAdapter));
239
240 return *this;
241
242 } catch (const std::exception& e) {
243
244 (void)e;
245 }
246
247 throw IllegalArgumentException("The provided data must be an instance of " \
248 "'BasicSignatureComputationDataAdapter' or " \
249 "'TraceableSignatureComputationDataAdapter'");

257 SamTransactionManager& prepareVerifySignature(const any data) override
258 {
259 /* C++: careful, code is a little bit different from Java because of any_cast flow */
260 try {
261
262 /* Basic signature */
263 auto dataAdapter =
264 any_cast<std::shared_ptr<BasicSignatureVerificationDataAdapter>>(data);
265
266 Assert::getInstance().notNull(dataAdapter, MSG_INPUT_OUTPUT_DATA)
267 .isInRange(dataAdapter->getData().size(),
268 1,
269 208,
270 "length of signed data to verify")
271 .isTrue(dataAdapter->getData().size() % 8 == 0,
272 "length of data to verify is a multiple of 8")
273 .isInRange(dataAdapter->getSignature().size(),
274 1,
275 8,
276 MSG_SIGNATURE_SIZE)
277 .isTrue(dataAdapter->isKeyDiversifierSet() == false ||
278 (dataAdapter->getKeyDiversifier().size() >= 1 &&
279 dataAdapter->getKeyDiversifier().size() <= 8),
280 MSG_KEY_DIVERSIFIER_SIZE_IS_IN_RANGE_1_8);
281
282 prepareSelectDiversifierIfNeeded(dataAdapter->getKeyDiversifier());
283 mSamCommands.push_back(
284 std::make_shared<CmdSamDataCipher>(mSam, nullptr, dataAdapter));
285
286 return *this;
287
288 } catch (const std::exception& e) {
289
290 (void)e;
291 }
292
293 try {
294
295 /* Traceable signature */
296 auto dataAdapter =
297 any_cast<std::shared_ptr<TraceableSignatureVerificationDataAdapter>>(data);
298
299 Assert::getInstance().notNull(dataAdapter, MSG_INPUT_OUTPUT_DATA)
300 .isInRange(dataAdapter->getData().size(),
301 1,
302 dataAdapter->isSamTraceabilityMode() ? 206 : 208,
303 "length of signed data to verify")
304 .isInRange(dataAdapter->getSignature().size(),
305 1,
306 8,
307 MSG_SIGNATURE_SIZE)
308 .isTrue(!dataAdapter->isSamTraceabilityMode() ||
309 (dataAdapter->getTraceabilityOffset() >= 0 &&
310 dataAdapter->getTraceabilityOffset() <=
311 static_cast<int>((dataAdapter->getData().size() * 8) -
312 (dataAdapter->isPartialSamSerialNumber() ?
313 7 * 8 : 8 * 8))),
314 "traceability offset is in range [0.." +
315 std::to_string((dataAdapter->getData().size() * 8) -
316 (dataAdapter->isPartialSamSerialNumber() ? 7 * 8 : 8 * 8)) +
317 "]")
318 .isTrue(dataAdapter->isKeyDiversifierSet() == false ||
319 (dataAdapter->getKeyDiversifier().size() >= 1 &&
320 dataAdapter->getKeyDiversifier().size() <= 8),
321 MSG_KEY_DIVERSIFIER_SIZE_IS_IN_RANGE_1_8);
322
323 /* Check SAM revocation status if requested. */
324 if (dataAdapter->isSamRevocationStatusVerificationRequested()) {
325
326 Assert::getInstance().notNull(mSecuritySetting, "security settings")
327 .notNull(mSecuritySetting->getSamRevocationServiceSpi(),
328 "SAM revocation service");
329
330 /* Extract the SAM serial number and the counter value from the data. */
331 const std::vector<uint8_t> samSerialNumber =
332 ByteArrayUtil::extractBytes(dataAdapter->getData(),
333 dataAdapter->getTraceabilityOffset(),
334 dataAdapter->isPartialSamSerialNumber() ? 3 : 4);
335
336 const int samCounterValue =
337 ByteArrayUtil::extractInt(
338 ByteArrayUtil::extractBytes(dataAdapter->getData(),
339 dataAdapter->getTraceabilityOffset() +
340 (dataAdapter->isPartialSamSerialNumber() ?
341 3 * 8 : 4 * 8),
342 3),
343 0,
344 3,
345 false);
346
347 /* Is SAM revoked ? */
348 if (mSecuritySetting->getSamRevocationServiceSpi()
349 ->isSamRevoked(samSerialNumber, samCounterValue)) {
350
351 throw SamRevokedException(
352 StringUtils::format("SAM with serial number '%s' and counter value '%d' " \
353 "is revoked.",
354 HexUtil::toHex(samSerialNumber).c_str(),
355 samCounterValue));
356 }
357 }
358
359 prepareSelectDiversifierIfNeeded(dataAdapter->getKeyDiversifier());
360 mSamCommands.push_back(
361 std::make_shared<CmdSamPsoVerifySignature>(mSam, dataAdapter));
362
363 return *this;
364
365 } catch (const std::bad_cast& e) {
366
367 /* C++: Fall through... */
368 (void)e;
369
370 } catch (const Exception& e) {
371
372 /* C++: rethrow since we are in a try/catch block */
373 (void)e;
374 throw;
375 }
376
377 throw IllegalArgumentException("The provided data must be an instance of " \
378 "'CommonSignatureVerificationDataAdapter'");

386 SamTransactionManager& processCommands() override
387 {
388 if (mSamCommands.empty()) {
389
390 return *this;
391 }
392
393 try {
394
395 /* Get the list of C-APDU to transmit */
396 const std::vector<std::shared_ptr<ApduRequestSpi>> apduRequests =
397 getApduRequests(mSamCommands);
398
399 /* Wrap the list of C-APDUs into a card request */
400 auto cardRequest = std::make_shared<CardRequestAdapter>(apduRequests, true);
401
402 /* Transmit the commands to the SAM */
403 const std::shared_ptr<CardResponseApi> cardResponse = transmitCardRequest(cardRequest);
404
405 /* Retrieve the list of R-APDUs */
406 const std::vector<std::shared_ptr<ApduResponseApi>>& apduResponses =
407 cardResponse->getApduResponses();
408
409
410 std::cout << "apduResponses size: " << apduResponses.size() << ", "
411 << "apduRequests size: " << apduRequests.size()
412 << std::endl;
413
414 /*
415 * If there are more responses than requests, then we are unable to fill the card image.
416 * In this case we stop processing immediately because it may be a case of fraud, and we
417 * throw an exception.
418 */
419 if (apduResponses.size() > apduRequests.size()) {
420
421 throw InconsistentDataException("The number of SAM commands/responses does not " \
422 "match: nb commands = " +
423 std::to_string(apduRequests.size()) +
424 ", nb responses = " +
425 std::to_string(apduResponses.size()) +
426 getTransactionAuditDataAsString());
427 }
428
429 /*
430 * We go through all the responses (and not the requests) because there may be fewer in
431 * the case of an error that occurred in strict mode. In this case the last response will
432 * raise an exception.
433 */
434 for (int i = 0; i < static_cast<int>(apduResponses.size()); i++) {
435
436 try {
437
438 mSamCommands[i]->parseApduResponse(apduResponses[i]);
439
440 } catch (const Exception& ex) {
441
442 /*
443 * C++: for some reason, it doesn't work if trying to catch directly the
444 * CalypsoSamCommandException exception.
445 */
446 const CalypsoSamCommandException &e =
447 dynamic_cast<const CalypsoSamCommandException&>(ex);
448
449 try {
450
451 const CalypsoSamCommand& commandRef =
452 std::dynamic_pointer_cast<AbstractSamCommand>(mSamCommands[i])
453 ->getCommandRef();
454
455 if (commandRef == CalypsoSamCommand::DIGEST_AUTHENTICATE) {
456
457 /* C++: cast made outside the if/else condition, will throw if false */
458 (void)static_cast<const CalypsoSamSecurityDataException&>(e);
459 throw InvalidCardSignatureException(
460 "Invalid card signature.",
461 std::make_shared<CalypsoSamCommandException>(e));
462
463 } else if (commandRef == CalypsoSamCommand::PSO_VERIFY_SIGNATURE ||
464 commandRef == CalypsoSamCommand::DATA_CIPHER) {
465
466 /* C++: cast made outside the if/else condition, will throw if false */
467 (void)static_cast<const CalypsoSamSecurityDataException&>(e);
468 throw InvalidSignatureException(
469 "Invalid signature.",
470 std::make_shared<CalypsoSamSecurityDataException>(
471 e.getMessage(),
472 dynamic_cast<const CalypsoSamCommand&>(e.getCommand()),
473 e.getStatusWord()));
474
475 } else if (commandRef == CalypsoSamCommand::SV_CHECK) {
476
477 /* C++: cast made outside the if/else condition, will throw if false */
478 (void)static_cast<const CalypsoSamSecurityDataException&>(e);
479 throw InvalidCardSignatureException(
480 "Invalid SV card signature.",
481 std::make_shared<CalypsoSamSecurityDataException>(
482 e.getMessage(),
483 dynamic_cast<const CalypsoSamCommand&>(e.getCommand()),
484 e.getStatusWord()));
485 }
486
487 } catch (const std::bad_cast& e) {
488
489 /* C++: Fall through... */
490 (void)e;
491
492 } catch (const Exception& e) {
493
494 /* C++: need to rethrow as we are in a try/catch block */
495 (void)e;
496 throw;
497 }
498
499 throw UnexpectedCommandStatusException(
500 MSG_SAM_COMMAND_ERROR +
501 "while processing responses to SAM commands: " +
502 e.getCommand().getName() +
503 getTransactionAuditDataAsString(),
504 std::make_shared<CalypsoSamCommandException>(e));
505 }
506 }
507
508 /*
509 * Finally, if no error has occurred and there are fewer responses than requests, then we
510 * throw an exception.
511 */
512 if (apduResponses.size() < apduRequests.size()) {
513
514 throw InconsistentDataException(
515 "The number of SAM commands/responses does not match:" \
516 " nb commands = " + std::to_string(apduRequests.size()) +
517 ", nb responses = " + std::to_string(apduResponses.size()) +
518 getTransactionAuditDataAsString());
519 }
520
521 } catch (const Exception& e) {
522
523 /* C++: need to rethrow as we are in a try/catch block */
524 (void)e;
525
526 /* Reset the list of commands (finally) */
527 mSamCommands.clear();
528
529 throw;
530 }
531
532 /* Reset the list of commands (finally) */
533 mSamCommands.clear();
534
535 return *this;
536 }

686 virtual std::shared_ptr<CardResponseApi> transmitCardRequest(
687 const std::shared_ptr<CardRequestSpi> cardRequest)
688 {
689 std::shared_ptr<CardResponseApi> cardResponse = nullptr;
690
691 try {
692 cardResponse = mSamReader->transmitCardRequest(cardRequest, ChannelControl::KEEP_OPEN);
693
694 } catch (const ReaderBrokenCommunicationException& e) {
695 saveTransactionAuditData(cardRequest, e.getCardResponse());
696 throw ReaderIOException(MSG_SAM_READER_COMMUNICATION_ERROR +
697 MSG_WHILE_TRANSMITTING_COMMANDS +
698 getTransactionAuditDataAsString(),
699 std::make_shared<ReaderBrokenCommunicationException>(e));
700
701 } catch (const CardBrokenCommunicationException& e) {
702 saveTransactionAuditData(cardRequest, e.getCardResponse());
703 throw SamIOException(MSG_SAM_COMMUNICATION_ERROR +
704 MSG_WHILE_TRANSMITTING_COMMANDS +
705 getTransactionAuditDataAsString(),
706 std::make_shared<CardBrokenCommunicationException>(e));
707
708 } catch (const UnexpectedStatusWordException& e) {
709 mLogger->debug("A SAM command has failed: %\n", e.getMessage());
710 cardResponse = e.getCardResponse();
711 }
712
713 saveTransactionAuditData(cardRequest, cardResponse);
714
715 return cardResponse;
716 }
